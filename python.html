import random personajes = ["kevin", "diana", "leo", "jose", "gabriel", "adan"] acciones = ["ataco", "salto", "corrio", "mato", "esquivo", "huyo"] lugares = ["en la cueva", "en la playa", "en la ciudad", "en la casa", "en la escuela", "en el bosque"] def cuento(): personaje = random.choice(personajes) accion = random.choice(acciones) lugar = random.choice(lugares) return f"{personaje} {accion} {lugar}" print(cuento())
